Programa: FWControlPanel.PRW
Classe FWControlPanel

Descrição:
Funcao cria um painel que permite manter a configuração de diversos itens 
Autor:
Ricardo Mansano
Release:
10.1.1.4
Data:
19/03/2009
Exemplo:

User Function TstPanel()
Local nWidthTree := 80
Local bOk := {|| alert('ok') }
Local bCancel := {|| alert('cancel') }
Local oConfig := FWControlPanel():New(180,180,550,700,,'Exemplo FWControlPanel',nWidthTree,bOk,bCancel)

 // Cria primeira sequencia de Itens
 oPanel01 := oConfig:AddTree( 'Painel 001','addcontainer',  )
 @30,10 SAY oSay VAR "Clique no Header" OF oPanel01 SIZE 50,10 PIXEL
 oCombo := TComboBox():New(40,10,,{'Item','Item'},73,10,oPanel01,,,,,,.T.,,,.F.,,.F.,,,,) 
 oConfig:EndTree()

 // Cria segunda sequencia de Itens
 oPanel02 := oConfig:AddTree( 'Painel 002','afastamento' )
 oGroup2 := TGroup():New(0,0,100,0,'Objeto TGroup 02',oPanel02,,,.T.)
 oGroup2:Align := CONTROL_ALIGN_TOP

 // Cria SubItens
 oPanel03 := oConfig:AddTreeItem( 'Painel 003',,{|opanel| alert('Entrou AddTreeItem: '+valtype(opanel)) } )
 oGroup3 := TGroup():New(0,0,100,0,'Objeto TGroup 03',oPanel03,,,.T.)
 oGroup3:Align := CONTROL_ALIGN_TOP

 oConfig:EndTree()
oConfig:Activate()
Return
